\section{Deployment}

\begin{frame}
    \frametitle{Real-Time Demo and Takeaways}
    \footnotesize
    \begin{columns}[t]
        \begin{column}{0.48\textwidth}
            \begin{block}{phyphox demo}
                \begin{itemize}
                    \item Polls Remote Access API every 0.05 s
                    \item Auto-detects time and acceleration buffers
                    \item Displays acceleration curves, detected steps, and cumulative count
                    \item Web dashboard available through Flask and Chart.js
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{exampleblock}{Key lessons}
                \begin{itemize}
                    \item Causality matters for live feedback
                    \item ACF is useful for lightweight fallback
                    \item Participant split exposes generalization gaps
                    \item Threshold and minimum interval convert probabilities into stable events
                \end{itemize}
            \end{exampleblock}
        \end{column}
    \end{columns}

    \begin{alertblock}{Limitations and next steps}
        TCN allows up to $\sim$180 steps/min with a 0.33 s interval, while ACF caps cadence at $\sim$168 steps/min. Future work: adaptive thresholds, placement adaptation, and hybrid ACF+TCN fallback.
    \end{alertblock}
\end{frame}
