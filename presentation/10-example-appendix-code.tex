\section{Deployment}

\begin{frame}
    \frametitle{Real-Time Demo and Validity}
    \footnotesize
    \begin{columns}[t]
        \begin{column}{0.48\textwidth}
            \begin{block}{phyphox demo}
                \begin{itemize}
                    \item Polls Remote Access API every 0.05 s
                    \item Auto-detects time and acceleration buffers
                    \item Displays acceleration curves, detected steps, and cumulative count
                    \item Web dashboard available through Flask and Chart.js
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{exampleblock}{Validity threats}
                \begin{itemize}
                    \item Only two self-recorded sessions
                    \item Subset evaluation is a stress test, not a full benchmark
                    \item Free-living step labels can have event-level timing noise
                    \item Fixed thresholds may not fit every gait regime
                \end{itemize}
            \end{exampleblock}
        \end{column}
    \end{columns}

    \begin{alertblock}{Limitations and next steps}
        TCN allows up to $\sim$180 steps/min with a 0.33 s interval, while ACF caps cadence at $\sim$168 steps/min. Next: adaptive thresholds, placement adaptation, short calibration walks, and hybrid ACF+TCN fallback.
    \end{alertblock}
\end{frame}
